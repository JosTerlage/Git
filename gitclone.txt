install salt using sh file

sudo salt-key --list-all

git clone https://github.com/JosTerlage/Git.git

sudo mkdir /srv/salt
sudo mv Git/masterinstall.sls /srv/salt
sudo mv Git/minioninstall.sls /srv/salt

cd Git/ && sudo chmod +x minioninstall.sh Nagiosinstallmaster.sh Saltstackinstallmaster.sh Saltstackinstallminion.sh Sysloginstallmaster.sh Nagiosinstallminion.sh Nagiosaddhost.sh && cd

sudo salt '*Master*' state.apply -t 2000 masterinstall
sudo salt '*Minion*' state.apply -t 2000 minioninstall
